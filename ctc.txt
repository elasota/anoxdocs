CTC model format:

0x00	char[4] magic			- "SMDL"
0x04	int version			- Always 11 in Anachronox data but code suggests it can be as low as 8
0x08	int skinWidth			- Skin width
0x0c	int skinHeight			- Skin height
0x10	int frameSize?
0x14	int numSkins
0x18	int numXYZ
0x1c	int numTexCoords
0x20	int numTris
0x24	int numGLCommands?
0x28	int numFrames
0x2c	int numAnimations
0x30	int numBones
0x34	int numMorphs
0x38	int numUnknown3
0x3c	int numUnknown4
0x40	int numUnknown5
0x44	float unknown6			- INTERNAL USE
0x48	int numUnknown7
0x4c	int offsetSkins			- Skin data (64 bytes)
0x50	int offsetVerts1		- Vert data of some sort, data is 6 dwords per xyz (24 bytes)
0x54	int offsetVerts2		- Vert data of some sort, data is 6 dwords per xyz (24 bytes)
0x58	int offsetTris1			- Tri data of some sort, data is 3 words per tri (6 bytes)
0x5c	int offsetTris2			- Tri data of some sort, data is 3 dwords per tri (12 bytes)
0x60	int offsetBones			- Bones?
0x64	int offsetSkeletalData		- See notes
0x68	int offsetAnimations
0x6c	int offsetFrames
0x70	int offsetGlCommands?
0x74	int offsetMorphData		- Vertex morphs
0x78	int offsetUnknown3		- Unknown3 data (12 bytes)
0x7c	int offsetUnknown4		- Unknown4 data (4 bytes)
0x80	int offsetUnknown5		- Unknown5 (28 bytes)
0x84	int offsetTris3			- Tri data of some sort, data is 40 bytes per tri, only present if numUnknown5 > 0
0x88	int offsetUnknown7		- Unknown data (1 byte)
0x8c	int eofPos

struct Bone
{
	int parentIndexPlusOne;
	int childCount;
	int childListPtr;		- INTERNAL USE
	
	float[38] unknown
}

struct SkinnedVertBoneWeight
{
	int bone
	float weight
}

struct SkinnedVert
{
	int numBones
	SkinnedVertBoneWeight[numBones] boneWeights
}

struct SkeletalData
{
	char[4] magic				- "STGW"
	SkinnedVert[model.numVerts] verts
}

struct FrameKey
{
	int unknown1
	int unknown2
	int unknown3
	int unknown4
	int unknown5
	int unknown6
}

struct Frame
{
	FrameKey[numFrames][numUnknownSize164 + 1] keys
}

struct FrameData
{
	Frame[numFrames] frames
}

struct VertMorph
{
	int unknown1
	int unknown2
	int unknown3
	int unknown4
}

struct Morph
{
	int unknown1
	int unknown2
	int unknown3
	int unknown4
	int numVertMorphs
	int unknown5
	int morphVertIndexPtr		- INTERNAL USE
	int morphDeltaPtr		- INTERNAL USE
}

struct MorphDynamicData
{
	int[numVertMorphs] xyzIndexes
	float[numVertMorphs][3] deltas
}


struct MorphData
{
	Morph[numMorphs] morphs
	MorphDynamicData[numMorphs] morphDynamicData
}

struct Unknown3
{
	char[8] name
	int unknown
}

struct Unknown4
{
	int unknown
}

struct Unknown5
{
	int unknown1
	int unknown2
	int unknown3		- Usually zero
	int unknown4		- Usually zero
	float[3] unknown5
}
